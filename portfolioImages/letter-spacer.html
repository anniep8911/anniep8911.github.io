  1. 프로그램 사용 목적

    레거시 코드에 대한 모든 파일 일괄적용 용도

  1. 프로그램 사용 시 기대 효과
    1. 700개 이상의 스타일파일에 대하여, 디자인 가이드 신규 변경건 일괄적용을 목표로 모든 글자크기당 조건을 파악하여 글자에대한 자간을 일괄적으로 수정하는 프로그램으로, 하루 8시간 기준 50개 파일(최대치)작업량 대비하여 200개 이상의 파일을 처리할 수 있음
    1. 불필요한 주석 자동 제거 및 더이상 사용하지 않는 css hack속성 제거

  1. 로직 워크플로우
    1. 디자인 가이드에 맞는 요구 사항 파악
    1. 폰트 사이즈가 발생할 수 있는 속성명 파악
    1. 불필요한 주석 타입의 형태 파악
    1. node활용, 스타일파일을 읽어내며 지워야 할 값들 리스트업 (letter-spacing, 불필요 주석, css hack)
    1. node를 활용하여 스타일 파일을 읽어낸 뒤 폰트 사이즈관련 속성명 뒤 바로 가이드에 맞는 레터 스페이싱 추가 한 뒤 배열로 저장
    1. 스트링타입으로 반환한뒤 다시 해당 파일에 덮어쓰기

  1. 프로그램 개발환경
    1. node : 14.17.0
    1. dependencies: none

  1. 프로그램의 효과

const jsDel=(dt,nowon)=>{
    dts = dt.split('\n');
    names.forEach(n=>{
        dts.forEach((e,i,a)=>{
            e.includes(n)?a[i]='':a[i]=e;
        });
    });

    dts= dts.map(m=> (m.includes('/*') && m.includes('*/') && !m.includes(':'))?!m.trim().includes('*/}')?(m.replace('**','*').replace('/*','//').replace('*/','')):m.replace('/*','//').replace('*/','}'):m);
    dts.forEach((e,i,a)=>{
        e.includes('/*') || e.includes('*/')?jsState= true:'';
        e.includes('/*')? jsState = true:'';
        e.includes('*/')? jsState = false:'';
        jsState===true||e.includes('*/')? a[i]='':a[i]=e; 
    });

    nowon ==='css'? dts= dts.map(m=> m.includes('//')&&!m.includes('htt')?m.replace('//','/*')+'*/':m):'';
    nowon === 'css'? dts = dts.filter(f=>f!=''):'';

    nowon === 'css'? dts = dts.map(m=> m.replace(/  /g,'')):'';
    ttDats =dts.map((m,i)=>(m.includes('//')&&!m.includes('http') &&!m.includes('svg')&&(m.includes(':')||m.includes('{')||m.includes('}')))?(m.search(reg)>0 || m.search(names)>0)?`${m}\n`:'':`${m}\n`).reduce((p,r)=>p+r);
    nowon === 'css'? ttDats = ttDats.toString().replace(/{\n/g,'{').replace(/;\n/g,';'):'';
    ttDats = ttDats.split('\n');
    nowon=='css'?ttDats = ttDats.map(m=> m.includes('root')?'':m):'';
    ttDats = ttDats.map(m=> m.includes('@charset')?m.replace(/;/,';\n'):m);
    ttDats = ttDats.map(m=>m.includes('{}') ||m.includes('{ }') ||m.includes('{\t}')?'':m+'\n');
    ttDats = ttDats.filter(f=>f!=='\n');
    return ttDats.join('').toString();
}
